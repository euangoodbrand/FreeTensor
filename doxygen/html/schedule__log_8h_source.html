86template <typename... Args>
87auto getIDFromPack(const std::tuple<Args...> &args) {
88 auto f = [&](auto &&arg) {
89 if constexpr (std::is_same_v<std::decay_t<decltype(arg)>,
90 IDMetadataPack>)
91 return arg.first;
92 else if constexpr (std::is_same_v<std::decay_t<decltype(arg)>,
93 std::vector<IDMetadataPack>>) {
94 auto ids =
95 arg | views::transform([&](auto pack) { return pack.first; });
96 return std::vector<ID>(ids.begin(), ids.end());
97 } else
98 return arg;
99 };
100 return std::apply(
101 [&](auto &&...args) { return std::make_tuple(f(args)...); }, args);
102}

122template <typename... Args>
123auto getPackFromID(auto schedule, const std::tuple<Args...> &args) {
124 auto f = [&](auto &&arg) {
125 if constexpr (std::is_same_v<std::decay_t<decltype(arg)>, ID>)
126 return IDMetadataPack{arg, schedule->find(arg)->metadata()};
127 else if constexpr (std::is_same_v<std::decay_t<decltype(arg)>,
128 std::vector<ID>>) {
129 auto packs =
130 arg | views::transform([&](auto id) {
131 return IDMetadataPack{id, schedule->find(id)->metadata()};
132 });
133 return std::vector<IDMetadataPack>(packs.begin(), packs.end());
134 } else
135 return arg;
136 };
137 return std::apply(
138 [&](auto &&...args) { return std::make_tuple(f(args)...); }, args);

193 std::lock_guard<std::mutex> guard(lock_);
194 if (std::holds_alternative<std::nullopt_t>(result_)) {
195 try {
196 result_ = std::apply(doSchedule_, getIDFromPack(params_));
197 } catch (...) {
198 result_ = std::current_exception();
199 }
200 }
201 }

208 ERROR("BUG: The schedule log is not run yet");

210 return std::get<Result>(result_);
211 } else {
