88template <typename... Args>
89auto getIDFromPack(const std::tuple<Args...> &args) {
90 auto f = [&](auto &&arg) {
91 if constexpr (std::is_same_v<std::decay_t<decltype(arg)>,
92 IDMetadataPack>)
93 return arg.first;
94 else if constexpr (std::is_same_v<std::decay_t<decltype(arg)>,
95 std::vector<IDMetadataPack>>) {
96 auto ids =
97 arg | views::transform([&](auto pack) { return pack.first; });
98 return std::vector<ID>(ids.begin(), ids.end());
99 } else
100 return arg;
101 };
102 return std::apply(
103 [&](auto &&...args) { return std::make_tuple(f(args)...); }, args);
104}

124template <typename... Args>
125auto getPackFromID(auto schedule, const std::tuple<Args...> &args) {
126 auto f = [&](auto &&arg) {
127 if constexpr (std::is_same_v<std::decay_t<decltype(arg)>, ID>)
128 return IDMetadataPack{arg, schedule->find(arg)->metadata()};
129 else if constexpr (std::is_same_v<std::decay_t<decltype(arg)>,
130 std::vector<ID>>) {
131 auto packs =
132 arg | views::transform([&](auto id) {
133 return IDMetadataPack{id, schedule->find(id)->metadata()};
134 });
135 return std::vector<IDMetadataPack>(packs.begin(), packs.end());
136 } else
137 return arg;
138 };
139 return std::apply(
140 [&](auto &&...args) { return std::make_tuple(f(args)...); }, args);

198 std::lock_guard<std::mutex> guard(lock_);
199 if (std::holds_alternative<std::nullopt_t>(result_)) {
200 try {
201 result_ = std::apply(doSchedule_, getIDFromPack(params_));
202 } catch (...) {
203 result_ = std::current_exception();
204 }
205 }
206 }

211 Result getResult() const {
212 if (std::holds_alternative<std::nullopt_t>(result_)) {
213 ERROR("BUG: The schedule log is not run yet");
214 } else if (std::holds_alternative<Result>(result_)) {
215 return std::get<Result>(result_);
216 } else {
217 ASSERT(std::holds_alternative<std::exception_ptr>(result_));
218 std::rethrow_exception(std::get<std::exception_ptr>(result_));
219 }
220 }

223 if (std::holds_alternative<std::nullopt_t>(result_)) {
224 ERROR("The schedule log is not run yet");
225 } else if (std::holds_alternative<Result>(result_)) {
226 Result result = std::get<Result>(result_);
227 if constexpr (std::derived_from<Result, Stmt>) {
228 return result;
229 } else {
230 return std::get<0>(result);
