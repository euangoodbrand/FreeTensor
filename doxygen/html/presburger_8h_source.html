935template <PBSetRef T> PBSet coefficients(T &&set, int64_t c = 0) {
936 // ISL coefficients applies Farkas lemma which disallows intermediate
937 // variables introduced by division/modulo constraints. The best we can do
938 // is to remove such constraints and provide a slightly relaxed bound for
939 // the coefficients, so remove_divs first.
940 auto coefficientsMap = isl_map_from_basic_map(isl_basic_set_unwrap(
941 isl_set_coefficients(isl_set_remove_divs(PBRefTake<T>(set)))));
942 auto ctx = isl_map_get_ctx(coefficientsMap);
943 auto paramsSpace = isl_space_domain(isl_map_get_space(coefficientsMap));
944 auto nParams = isl_space_dim(paramsSpace, isl_dim_set);
945 auto cPoint = isl_point_zero(paramsSpace);
946 isl_point_set_coordinate_val(cPoint, isl_dim_set, nParams - 1,
947 isl_val_int_from_si(ctx, -c));
948 return apply(PBSet(isl_set_from_point(cPoint)), PBMap(coefficientsMap));
949}
