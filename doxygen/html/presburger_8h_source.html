25template <class T> T *GET_ISL_PTR(T *ptr) {
26 ASSERT(ptr != nullptr);
27 return ptr;
28}
29
30#define COPY_ISL_PTR(ptr, type) _COPY_ISL_PTR(ptr, isl_##type##_copy)
31template <class T> T *_COPY_ISL_PTR(const T *ptr, T *(copy)(T *)) {
32 ASSERT(ptr != nullptr);
33 return copy(const_cast<T *>(ptr));
34}
35
36template <class T> T *MOVE_ISL_PTR(T *&ptr) {
37 ASSERT(ptr != nullptr);
38 auto ret = ptr;
39 ptr = nullptr;
40 return ret;
41}

655 auto coefficientsMap = isl_map_from_basic_map(
656 isl_basic_set_unwrap(isl_set_coefficients(PBRefTake<T>(set))));
657 auto ctx = isl_map_get_ctx(coefficientsMap);

660 auto cPoint = isl_point_zero(paramsSpace);
