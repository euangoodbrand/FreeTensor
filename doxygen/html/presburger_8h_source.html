23template <class T> T *GET_ISL_PTR(T *ptr) {
24 ASSERT(ptr != nullptr);
25 return ptr;
26}
27
28#define COPY_ISL_PTR(ptr, type) _COPY_ISL_PTR(ptr, isl_##type##_copy)
29template <class T> T *_COPY_ISL_PTR(const T *ptr, T *(copy)(T *)) {
30 ASSERT(ptr != nullptr);
31 return copy(const_cast<T *>(ptr));
32}
33
34template <class T> T *MOVE_ISL_PTR(T *&ptr) {
35 ASSERT(ptr != nullptr);
36 auto ret = ptr;
37 ptr = nullptr;
38 return ret;
39}
