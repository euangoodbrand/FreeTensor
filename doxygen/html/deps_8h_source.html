164 template <class T> void visitStoreLike(const T &op) {
165 BaseClass::visit(op);
166
167 bool isThisVarDef = false;
168 VarDef viewOf;
169 if (def(op->var_)->id() == vardef_) {
170 isThisVarDef = true;
171 } else {
172 for (auto source = def(op->var_); source->viewOf_.has_value();) {
173 source = def(*source->viewOf_);
174 if (source->id() == vardef_) {
175 isThisVarDef = true;
176 viewOf = source;
177 break;
178 }
179 }
180 }
181 if (!isThisVarDef) {
182 return;
183 }
184
185 auto old = cur_;
186 auto oldLastIsLoad = lastIsLoad_;
187 if (!cur_.empty() &&
188 cur_.back().iter_->nodeType() == ASTNodeType::IntConst) {
189 // top is band node
190 cur_.back().iter_ =
191 makeIntConst(cur_.back().iter_.as<IntConstNode>()->val_ + 1);
192 }
193 lastIsLoad_ = false;
194
195 std::vector<Expr> exprs;
196 VarDef d;
197 if (viewOf.isValid()) {
198 // Simultaneously access of a `VarDef` and the `VarDef` it views is
199 // ALWAYS treated as dependences. Use Intrinsic as "any expression"
200 exprs = std::vector<Expr>(
201 viewOf->buffer_->tensor()->shape().size(),
202 makeIntrinsic("", {}, DataType::Int32, false));
203 d = viewOf;
204 } else {
205 exprs = normalizeExprs(op->indices_, op->id());
206 d = def(op->var_);
207 }
208
209 auto ap = Ref<AccessPoint>::make();
210 *ap = {op, curStmt(), d, d->buffer_, defAxis_,
211 cur_, std::move(exprs), conds_};
212 if (accFilter_ == nullptr || accFilter_(*ap)) {
213 subTreeFilteredIn_.insert(op);
214 writes_.emplace_back(ap);
215 } else {
216 // No stepping to make iteration space more compact
217 cur_ = std::move(old);
218 lastIsLoad_ = oldLastIsLoad;
219 }
220 }
