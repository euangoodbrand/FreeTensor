34 template <std::uniform_random_bit_generator RNG>
35 std::vector<double> dirichletSample(RNG &rng) const {
36 // Compute Dirichlet distribution via Gamma distribution
37 // (https://en.wikipedia.org/wiki/Dirichlet_distribution#Computational_methods)
38 auto n = obs_.size();
39 std::vector<double> ret(n);
40 for (size_t i = 0; i < n; i++) {
41 ret[i] = std::gamma_distribution<double>{obs_[i] + 1., 1.}(rng);
42 }
43 // std::discrete_distribution will normalize it for us
44 return ret;
45 }

47 public:
48 DiscreteRandVar(const std::string &name, const RandCondStack &conds,
49 const std::vector<int> &initObs)
50 : name_(name), conds_(conds), obs_(initObs) {}
51
52 void observe(int value, int cnt = 1) { obs_.at(value) += cnt; }
53
54 template <std::uniform_random_bit_generator RNG>
55 int sample(RNG &rng) const {
56 auto discreteProb = dirichletSample(rng);
57 return std::discrete_distribution<int>(discreteProb.begin(),
58 discreteProb.end())(rng);
59 }
