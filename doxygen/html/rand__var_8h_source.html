34 // Compute Dirichlet distribution via Gamma distribution
35 // (https://en.wikipedia.org/wiki/Dirichlet_distribution#Computational_methods)
36 auto n = obs_.size();
37 std::vector<double> ret(n);

39 ret[i] = std::gamma_distribution<double>{obs_[i] + 1., 1.}(rng);
40 }
