114template <class Simplifier> Stmt simplifyImpl(const Stmt &_op) {
115 auto op = _op;
116
117 for (int i = 0;; i++) {
118 op = annotateConds(op);
119 auto newOp = Simplifier()(op);
120 if (HashComparator()(newOp, op) || i > 100) {
121 if (i > 100) {
122 WARNING("SimplifyPass iterates over 100 rounds. Maybe there is "
123 "a bug");
124 }
125 return newOp;
126 }
127 op = newOp;
128 }
129}
