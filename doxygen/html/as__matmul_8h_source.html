47 for (auto &&[idx, dimLen, baseIdx] :
48 views::zip(acc->indices_, buffer(acc->var_)->tensor()->shape(),

50 auto &&lin = analyzeLinear(idx);
51 if (lin.coeff_.size() != 1 ||
52 std::abs(lin.coeff_.front().k_) != 1 ||

58 continue; // not a dim in matmul

60 Var var = lin.coeff_.front().a_.template as<VarNode>();
61 if (!iterMap_.count(var->name_)) {
62 continue; // not a dim in matmul
63 } else {
64 baseIdx = makeIntConst(0);
65 }
66 int loopLevel = iterMap_.at(var->name_);
67 if (!HashComparator()(loop(var->name_)->len_, dimLen)) {
68 throw InvalidSchedule(

70 " should loop over the entire range (" + toString(dimLen) +
71 "), instead of " + toString(loop(var->name_)->len_));
72 }
73 usedBy[loopLevel] = true;
74 order.emplace_back(loopLevel);
75 }

93 thisDimIn = false;

96 if (!iterMap_.count(var->name_) ||
97 !flag[iterMap_.at(var->name_)]) {
98 thisDimIn = false;
99 }
100 }
101 if (thisDimIn) {
