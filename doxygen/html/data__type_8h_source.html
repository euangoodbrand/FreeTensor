35inline BaseDataType parseBaseDataType(const std::string &_str) {
36 auto &&str = tolower(_str);
37 for (auto &&[i, s] : views::enumerate(baseDataTypeNames)) {
38 if (s == str) {
39 return (BaseDataType)i;
40 }
41 }
42 std::string msg = "Unrecognized base data type \"" + _str +
43 "\". Candidates are (case-insensitive): ";
44 for (auto &&[i, s] : views::enumerate(baseDataTypeNames)) {
45 msg += (i > 0 ? ", " : "");
46 msg += s;
47 }
48 ERROR(msg);
