55 template <class T> void visitMemAcc(const T &op) {
56 BaseClass::visit(op);
57
58 auto mtype = buffer(op->var_)->mtype();
59 if (mtype != MemType::GPULocal && mtype != MemType::GPUShared &&
60 mtype != MemType::GPUWarp) {
61 return;
62 }
63
64 auto &&defId = def(op->var_)->id();
65 std::vector<Ref<SimplexOffset>> thisOffsets;
66 for (auto &&idx : op->indices_) {
67 Ref<SimplexOffset> offset;
68 if (mtype == MemType::GPUShared || mtype == MemType::GPUWarp) {
69 offset =
70 getSimplexOffset({blockIdxX, blockIdxY, blockIdxZ}, idx);
71 } else {
72 offset = getSimplexOffset({threadIdxX, threadIdxY, threadIdxZ,
73 blockIdxX, blockIdxY, blockIdxZ},
74 idx);
75 }
76 thisOffsets.emplace_back(offset);
77 }
78
79 if (!offsets_.count(defId)) {
80 offsets_[defId] = thisOffsets;
81 } else {
82 ASSERT(offsets_.at(defId).size() == thisOffsets.size());
83 for (auto &&[old, cur] :
84 iter::zip(offsets_.at(defId), thisOffsets)) {
85 if (old.isValid() &&
86 (!cur.isValid() || old->offset_ != cur->offset_)) {
87 old = nullptr;
88 }
89 }
90 }
