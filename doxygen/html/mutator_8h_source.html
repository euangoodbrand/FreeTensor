264 virtual Stmt visit(const For &op) {
265 auto begin = (*this)(op->begin_);
266 auto end = (*this)(op->end_);

268 auto len = (*this)(op->len_);
269 auto property = Ref<ForProperty>::make()
270 ->withParallel(op->property_->parallel_)
271 ->withUnroll(op->property_->unroll_)
272 ->withVectorize(op->property_->vectorize_)
273 ->withNoDeps(op->property_->noDeps_)
274 ->withPreferLibs(op->property_->preferLibs_);
275 property->reductions_.reserve(op->property_->reductions_.size());
276 for (auto &&r : op->property_->reductions_) {
277 std::vector<Expr> begins, ends;
278 begins.reserve(r->begins_.size());
279 ends.reserve(r->ends_.size());
280 for (auto &&item : r->begins_) {
281 begins.emplace_back((*this)(item));
282 }
283 for (auto &&item : r->ends_) {
284 ends.emplace_back((*this)(item));
285 }
286 property->reductions_.emplace_back(makeReductionItem(
287 r->op_, r->var_, std::move(begins), std::move(ends)));
288 }
289 auto body = (*this)(op->body_);
290 auto ret = makeFor(op->iter_, std::move(begin), std::move(end),
291 std::move(step), std::move(len), std::move(property),
292 std::move(body), op->metadata(), op->id());
