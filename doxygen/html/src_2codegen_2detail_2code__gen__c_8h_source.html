19template <class Stream>
20std::function<std::ostream &(std::ostream &)>
21CodeGenC<Stream>::genMdPtrType(const VarDef &def, bool isConst) {
22 // NOTE: `[=]` implicitly capturing `this` is deprecated in C++20, but if we
23 // use `[=, this]`, clang will raise a warning because it will think `this`
24 // is duplicated.
25 return [=](std::ostream &os) -> std::ostream & {
26 auto &&buf = def->buffer_;
27
28 if (buf->tensor()->shape().empty()) {
29 // Use reference for scalars
30 if (isConst) {
31 os << "const ";
32 }
33 os << gen(buf->tensor()->dtype()) << " &";
34 return os;
35 }
36
37 bool isRestricted = true;
38 if (def->viewOf_.has_value() ||
39 !findAllStmt(def, [&](const Stmt &inner) {
40 return inner->nodeType() == ASTNodeType::VarDef &&
41 inner.as<VarDefNode>()->viewOf_ == def->name_;
42 }).empty()) {
43 isRestricted = false;
44 }
45
46 os << (Config::debugRuntimeCheck() ? "mdspan_dbg<"
47 : isRestricted ? "mdspan_r<"
48 : "mdspan<");
49 if (isConst) {
50 os << "const ";
51 }
52 os << gen(buf->tensor()->dtype()) << ", extents<";
53 for (auto &&[i, dim] : views::enumerate(buf->tensor()->shape())) {
54 os << (i > 0 ? ", " : "");
55 if (dim->nodeType() == ASTNodeType::IntConst) {
56 os << dim.template as<IntConstNode>()->val_;
57 } else {
58 os << "dynamic_extent";
59 }
60 }
61 return os << ">>";
62 };
63}

65template <class Stream>
66void CodeGenC<Stream>::genMdPtrDef(const VarDef &def,
67 const std::function<void()> &genRawPtr,
68 bool isConst) {
69 auto &&buf = def->buffer_;
70
71 if (buf->tensor()->shape().empty()) {
72 // Use reference for scalars
73 // e.g.
74 // ((const int32_t &)*((const int32_t *)(...)))
75 this->os() << "((" << genMdPtrType(def, isConst) << ")*((";
76 if (isConst) {
77 this->os() << "const ";
78 }
79 this->os() << gen(buf->tensor()->dtype()) << " *)(";
80 genRawPtr();
81 this->os() << ")))";
82 return;
83 }
84
85 this->os() << genMdPtrType(def, isConst) << "((";
86 if (isConst) {
87 this->os() << "const ";
88 }
89 this->os() << gen(buf->tensor()->dtype()) << "*)(";
90 genRawPtr();
91 this->os() << ")";
92 for (auto &&dim : buf->tensor()->shape()) {
93 if (dim->nodeType() != ASTNodeType::IntConst) {
94 this->os() << ", ";
95 (*this)(dim);
96 }
97 }
98 this->os() << ")";
99}

104 if (def->buffer_->mtype() == MemType::ByValue) {

120 }

139template <class Stream> void CodeGenC<Stream>::visit(const VarDef &op) {
140 this->makeIndent();
141 auto &&tensor = op->buffer_->tensor();
142 auto &&shape = tensor->shape();
143 auto name = mangle(op->name_);
144
145 if (op->viewOf_.has_value()) {
146 // e.g.
147 // auto &&x = mdspan_r<const float, extents<5, 5>>(y.data_handle());
148 auto source = op;
149 while (source->viewOf_.has_value()) {
150 source = this->def(*source->viewOf_);
151 }
152 this->os() << "auto &&" << name << " = ";
153 genMdPtrDef(op, mangle(source->name_) + ".data_handle()",
154 !isWritable(source->buffer_->atype()));
155 this->os() << ";" << std::endl;
156
157 } else if (!isInputting(op->buffer_->atype()) &&
158 !isOutputting(op->buffer_->atype())) {
159 // e.g. 1. float x;
160 // 2. float x[5][5][5];
161 this->os() << gen(tensor->dtype()) << " " << name;
162 for (auto &&dim : shape) {
163 this->os() << "[";
164 (*this)(dim);
165 this->os() << "]";
166 }
167 this->os() << ";" << std::endl;
168 } else {
169 auto paramPositions = ranges::to<std::vector>(
170 params_ | views::enumerate | views::filter([&](auto &&pair) {
171 return pair.second.name_ == op->name_;
172 }) |
173 views::keys);
174 auto returnPositions = ranges::to<std::vector>(
175 returns_ | views::enumerate | views::filter([&](auto &&pair) {
176 return pair.second.name_ == op->name_;
177 }) |
178 views::keys);
179 bool isParam = !paramPositions.empty();
180 bool isReturn = !returnPositions.empty();
181 if (!isParam && !isReturn) {
182 throw InvalidProgram("I/O variable " + op->name_ +
183 " used but not defined as a function's "
184 "parameters or return values");
185 }
186 std::string rawPtr;
187 if (isParam) {
188 if (paramPositions.size() > 1) {
189 throw InvalidProgram("Parameter '" + op->name_ +
190 "' is duplicated");
191 }
192 int nthParam = paramPositions.front();
193 rawPtr = "params[" + std::to_string(nthParam) + "]";
194 } else {
195 if (!isOutputting(op->buffer_->atype())) {
196 throw InvalidProgram(
197 "Only outputting variable can be as a return value");
198 }
199 // If there are multiple position with the same name, we only fill
200 // the first position. Driver::collectReturns will only collect the
201 // first
202 int nthReturn = returnPositions.front();
203 rawPtr = "returns[" + std::to_string(nthReturn) + "]";
204 std::string shapePtr =
205 "retShapes[" + std::to_string(nthReturn) + "]";
206 std::string dimPtr = "retDims[" + std::to_string(nthReturn) + "]";
207 this->os() << "if (" + rawPtr + " == NULL) ";
208 this->beginBlock();
209 this->genAlloc(op->buffer_->tensor(), rawPtr, shapePtr, dimPtr);
210 this->endBlock();
211 this->makeIndent();
212 }
213
214 switch (op->buffer_->mtype()) {
215 case MemType::ByValue:
216 // e.g. (1)
217 // float x;
218 // x = *((float*)params[0]);
219
220 // e.g. (2)
221 // __ByValArray<__ByValArray<float, 2>, 2> x;
222 // x[0][0] = *((float*)params[0])[0];
223 // x[0][1] = *((float*)params[0])[1];
224 // x[1][0] = *((float*)params[0])[2];
225 // x[1][1] = *((float*)params[0])[3];
226 if (op->buffer_->atype() != AccessType::Input) {
227 throw InvalidProgram("ByValue typed var " + op->name_ +
228 " can only be Input");
229 }
230 for (auto &&dim : shape) {
231 if (dim->nodeType() != ASTNodeType::IntConst) {
232 throw InvalidProgram("ByValue typed var " + op->name_ +
233 " can only have a constant size");
234 }
235 }
236 if (shape.empty()) {
237 this->os() << gen(tensor->dtype()) << " " << name << " = *(("
238 << gen(tensor->dtype()) << "*)" << rawPtr << ");"
239 << std::endl;
240 } else {
241 for (size_t i = 0, iEnd = shape.size(); i < iEnd; i++) {
242 this->os() << "__ByValArray<";
243 }
244 this->os() << gen(tensor->dtype());
245 for (auto it = shape.rbegin(); it != shape.rend(); it++) {
246 this->os() << ", " << (*it).as<IntConstNode>()->val_ << ">";
247 }
248 this->os() << " " << name << ";" << std::endl;
249 std::vector<int> idx(shape.size(), 0);
250 std::function<void(size_t, int)> f = [&](size_t i, int offset) {
251 if (i == shape.size()) {
252 this->makeIndent();
253 this->os() << name;
254 for (int x : idx) {
255 this->os() << "[" << x << "]";
256 }
257 this->os()
258 << " = ((" << gen(tensor->dtype()) << "*)" << rawPtr
259 << ")[" << offset << "];" << std::endl;
260 return;
261 }
262 for (int j = 0, jEnd = shape[i].as<IntConstNode>()->val_;
263 j < jEnd; j++) {
264 idx[i] = j;
265 f(i + 1, offset * jEnd + j);
266 }
267 };
268 f(0, 0);
269 }
270 break;
271
272 default:
273 // e.g.
274 // auto &&x = mdspan_r<const float, extents<5, 5>>(params[0]);
275 this->os() << "auto &&" << name << " = ";
276 genMdPtrDef(op, rawPtr, !isWritable(op->buffer_->atype()));
277 this->os() << ";" << std::endl;
278 }
279 }
280
281 this->markDef(op);
282 (*this)(op->body_);
283 this->markUndef(op);
284}

351template <class Stream> void CodeGenC<Stream>::visit(const ReduceTo &op) {
352 this->markUse(op->var_);
353
354 this->makeIndent();
355
356 auto genAddr = [&]() { this->genScalar(op); };
357 auto genExpr = [&]() { (*this)(op->expr_); };
358
359 switch (op->op_) {
360 case ReduceOp::Add:
361 genAddr(), this->os() << " += ", genExpr();
362 break;
363 case ReduceOp::Mul:
364 genAddr(), this->os() << " *= ", genExpr();
365 break;
366 case ReduceOp::Min:
367 genAddr(), this->os()
368 << " = std::min<"
369 << this->gen(this->buffer(op->var_)->tensor()->dtype())
370 << ">(";
371 genAddr(), this->os() << ", ", genExpr(), this->os() << ")";
372 break;
373 case ReduceOp::Max:
374 genAddr(), this->os()
375 << " = std::max<"
376 << this->gen(this->buffer(op->var_)->tensor()->dtype())
377 << ">(";
378 genAddr(), this->os() << ", ", genExpr(), this->os() << ")";
379 break;
380 case ReduceOp::LAnd:
381 genAddr(), this->os() << " &= (bool)(", genExpr(), this->os() << ")";
382 break;
383 case ReduceOp::LOr:
384 genAddr(), this->os() << " |= (bool)(", genExpr(), this->os() << ")";
385 break;
386 default:
387 ASSERT(false);
388 }
389
390 this->os() << ";" << std::endl;
391}
