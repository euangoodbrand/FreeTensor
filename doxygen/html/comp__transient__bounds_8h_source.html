67 void applyCond(const Expr &_cond,
68 const std::unordered_set<std::string> &bodyAllWrites) {
69 auto dnf = asDNF(_cond);
70
71 if (dnf.size() != 1) {
72 return; // Currently we cannot handle OR
73 }
74
75 for (auto &&cond : dnf.front()) {
76 if (hasIntersect(allReads(cond), bodyAllWrites)) {
77 continue;
78 }
79
80 auto norm = linearComp(cond);
81 if (!norm.has_value()) {
82 continue;
83 }
84
85 auto &&[lin, type] = *norm;
86 if (!isInt(lin2expr(lin)->dtype())) {
87 continue;
88 }
89
90 for (auto &&[k, a] : lin.coeff_) {
91 if (a->nodeType() == ASTNodeType::Var ||
92 a->nodeType() == ASTNodeType::Load) {
93 auto [lower, upper] = lin2bounds(lin, type, a);
94 if (lower.has_value()) {
95 transients_[a].expr_ = a;
96 transients_[a].lower_.emplace_back(lower->expr());
97 }
98 if (upper.has_value()) {
99 transients_[a].expr_ = a;
100 transients_[a].upper_.emplace_back(upper->expr());
101 }
102 }
103 }
104
105 conds_.emplace_back(cond);
106 }

124 if (op->step_->nodeType() == ASTNodeType::IntConst) {
125 auto step = op->step_.as<IntConstNode>()->val_;
126 if (step > 0) {
127 transients_[var] = {
128 var, {op->begin_}, {makeSub(op->end_, makeIntConst(1))}};
129 conds_.emplace_back(makeGE(var, op->begin_));

131 conds_.emplace_back(
132 makeEQ(makeMod(makeSub(var, op->begin_), op->step_),
133 makeIntConst(0)));
134 } else if (step < 0) {

137 conds_.emplace_back(makeLE(var, op->begin_));

139 conds_.emplace_back(
140 makeEQ(makeMod(makeSub(var, op->begin_), op->step_),
141 makeIntConst(0)));
142 } else {
143 transients_[var] = {var, {op->begin_}, {op->begin_}};
144 conds_.emplace_back(makeEQ(var, op->begin_));
145 }
146 }
147 this->pushFor(op);
148 MAYBE_VOID(body, (*this)(op->body_));
149 this->popFor(op);
150 conds_.resize(oldCondsSize);

152
153 if constexpr (!std::is_same_v<typename BaseClass::StmtRetType, void>) {
