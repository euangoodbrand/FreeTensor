48template <class BaseClass>
49class CompTransientBounds : public BaseClass,
50 public CompTransientBoundsInterface {
51 // Bounds related to certain expressions
52 // Bounds in transients_ has already been recursed with (*this)(...)
53 ASTHashMap<Expr, TransientBound> transients_;
54
55 // Original bounds
56 std::vector<Expr> conds_;
57
58 public:
59 TransientBound transient(const Expr &op) const override {
60 if (transients_.count(op)) {
61 return transients_.at(op);
62 }
63 return {};
64 }
65
66 const std::vector<Expr> &conds() const override { return conds_; }
67
68 private:
69 void applyCond(const Expr &_cond,
70 const std::unordered_set<std::string> &bodyAllWrites) {
71 auto dnf = asDNF(_cond);
72
73 if (dnf.size() != 1) {
74 return; // Currently we cannot handle OR
75 }
76
77 for (auto &&cond : dnf.front()) {
78 if (cond->nodeType() == ASTNodeType::Unbound) {
79 continue;
80 }
81
82 if (hasIntersect(allReads(cond), bodyAllWrites)) {
83 continue;
84 }
85
86 auto norm = linearComp(cond);
87 if (!norm.has_value()) {
88 continue;
89 }
90
91 auto &&[lin, type] = *norm;
92 if (!isInt(lin2expr(lin)->dtype())) {
93 continue;
94 }
95
96 for (auto &&[k, a] : lin.coeff_) {
97 if (a->nodeType() == ASTNodeType::Var ||
98 a->nodeType() == ASTNodeType::Load) {
99 auto [lower, upper] = lin2bounds(lin, type, a);
100 if (lower.has_value()) {
101 transients_[a].expr_ = a;
102 transients_[a].lower_.emplace_back(lower->expr());
103 }
104 if (upper.has_value()) {
105 transients_[a].expr_ = a;
106 transients_[a].upper_.emplace_back(upper->expr());
107 }
108 }
109 }
110
111 conds_.emplace_back(cond);
112 }
113 }

116 using BaseClass::visit; // Avoid hiding virtual functions

118 typename BaseClass::StmtRetType visit(const For &op) override {
119 MAYBE_VOID(begin, (*this)(op->begin_));
120 MAYBE_VOID(end, (*this)(op->end_));
121 MAYBE_VOID(step, (*this)(op->step_));
122 MAYBE_VOID(len, (*this)(op->len_));
123
124 auto var = makeVar(op->iter_);
125 if (transients_.count(var)) {
126 throw InvalidProgram(
127 "iterators with the same name in nested loops are not allowed");
128 }
129 auto oldCondsSize = conds_.size();
130 if (op->step_->nodeType() == ASTNodeType::IntConst) {
131 auto step = op->step_.as<IntConstNode>()->val_;
132 if (step > 0) {
133 transients_[var] = {
134 var, {op->begin_}, {makeSub(op->end_, makeIntConst(1))}};
135 conds_.emplace_back(makeGE(var, op->begin_));
136 conds_.emplace_back(makeLT(var, op->end_));
137 conds_.emplace_back(
138 makeEQ(makeMod(makeSub(var, op->begin_), op->step_),
139 makeIntConst(0)));
140 } else if (step < 0) {
141 transients_[var] = {
142 var, {makeAdd(op->end_, makeIntConst(1))}, {op->begin_}};
143 conds_.emplace_back(makeLE(var, op->begin_));
144 conds_.emplace_back(makeGT(var, op->end_));
145 conds_.emplace_back(
146 makeEQ(makeMod(makeSub(var, op->begin_), op->step_),
147 makeIntConst(0)));
148 } else {
149 transients_[var] = {var, {op->begin_}, {op->begin_}};
150 conds_.emplace_back(makeEQ(var, op->begin_));
151 }
152 }
153 if constexpr (std::is_base_of_v<SymbolTableInterface, BaseClass>) {
154 this->pushFor(op);
155 }
156 MAYBE_VOID(body, (*this)(op->body_));
157 if constexpr (std::is_base_of_v<SymbolTableInterface, BaseClass>) {
158 this->popFor(op);
159 }
160 conds_.resize(oldCondsSize);
161 transients_.erase(var);
162
163 if constexpr (!std::is_same_v<typename BaseClass::StmtRetType, void>) {
164 auto property = Ref<ForProperty>::make()
165 ->withParallel(op->property_->parallel_)
166 ->withUnroll(op->property_->unroll_)
167 ->withVectorize(op->property_->vectorize_)
168 ->withNoDeps(op->property_->noDeps_)
169 ->withPreferLibs(op->property_->preferLibs_);
170 property->reductions_.reserve(op->property_->reductions_.size());
171 for (auto &&r : op->property_->reductions_) {
172 std::vector<Expr> begins, ends;
173 begins.reserve(r->begins_.size());
174 ends.reserve(r->ends_.size());
175 for (auto &&item : r->begins_) {
176 begins.emplace_back((*this)(item));
177 }
178 for (auto &&item : r->ends_) {
179 ends.emplace_back((*this)(item));
180 }
181 property->reductions_.emplace_back(
182 makeReductionItem(r->op_, r->var_, std::move(begins),
183 std::move(ends), r->syncFlush_));
184 }
185 return makeFor(op->iter_, std::move(begin), std::move(end),
186 std::move(step), std::move(len), std::move(property),
187 std::move(body), op->metadata(), op->id(),
188 op->debugBlame());
189 }
190 }

192 typename BaseClass::StmtRetType visit(const If &op) override {
193 MAYBE_VOID(cond, (*this)(op->cond_));
194
195 auto oldMap = transients_;
196 auto oldCondsSize = conds_.size();
197 applyCond(op->cond_, allWrites(op->thenCase_));
198 MAYBE_VOID(thenCase, (*this)(op->thenCase_));
199 transients_ = oldMap;
200 conds_.resize(oldCondsSize);
201

203 if (op->elseCase_.isValid()) {
204 auto oldCondsSize = conds_.size();
205 applyCond(makeLNot(op->cond_), allWrites(op->elseCase_));
206 MAYBE_VOID_ASSIGN(elseCase, (*this)(op->elseCase_));
207 transients_ = oldMap;
208 conds_.resize(oldCondsSize);
209 }
210
211 if constexpr (!std::is_same_v<typename BaseClass::StmtRetType, void>) {
212 return makeIf(std::move(cond), std::move(thenCase),

215 }
