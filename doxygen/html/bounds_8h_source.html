74std::pair<Opt<LowerBound>, Opt<UpperBound>>
75lin2bounds(const LinearExpr<T> &_lin, ASTNodeType cmp, const Expr &x) {
76 typedef std::pair<Opt<LowerBound>, Opt<UpperBound>> RetType;
77
78 // 1. Remove x from lin
79 // 2. Convert to a rational linear because we need to do division later
80 LinearExpr<Rational<int64_t>> lin;
81 Opt<Rational<int64_t>> selfK;
82 lin.bias_ = _lin.bias_;
83 lin.coeff_.reserve(_lin.coeff_.size() - 1);
84 for (auto &&[k, a] : _lin.coeff_) {
85 if (HashComparator()(a, x)) {
86 ASSERT(!selfK.isValid());
87 selfK = Opt<Rational<int64_t>>::make(k);
88 } else {
89 lin.coeff_.emplace_back(Scale<Rational<int64_t>>{k, a});
90 }
91 }
92 if (!selfK.isValid() || *selfK == 0) {
93 return RetType(nullptr, nullptr);
94 }
95
96 // 3. Normalize according to selfK
97 // Now x is at the left side and the other items are at the right side
98 if (*selfK < 0) {
99 cmp = detail::reverseCmp(cmp);
100 }
101 lin.bias_ /= -*selfK;
102 for (auto &item : lin.coeff_) {
103 item.k_ /= -*selfK;
104 }
105
106 // 4. Construct the bounds according to cmp
107 // We normalize LT and GT to LE and GE according to the following:
108 // selfK * x < y <==> selfK * x <= y - 1 (as an integer expression)
109 // x < 1/selfK * y <==> x <= 1/selfK * y - 1/selfK
110 switch (cmp) {
111 case ASTNodeType::LE:
112 return RetType(nullptr, Opt<UpperBound>::make(lin));
113 case ASTNodeType::LT:
114 return RetType(nullptr,
115 Opt<UpperBound>::make(LinearExpr<Rational<int64_t>>{
116 lin.coeff_, lin.bias_ - 1 / std::abs(*selfK)}));
117 case ASTNodeType::GE:
118 return RetType(Opt<LowerBound>::make(lin), nullptr);
119 case ASTNodeType::GT:
120 return RetType(Opt<LowerBound>::make(LinearExpr<Rational<int64_t>>{
121 lin.coeff_, lin.bias_ + 1 / std::abs(*selfK)}),
122 nullptr);
123 case ASTNodeType::EQ:
124 return RetType(Opt<LowerBound>::make(lin), Opt<UpperBound>::make(lin));
125 default:
126 return RetType(nullptr, nullptr);
127 }
128}
