599 case ASTNodeType::Sub:

601 case ASTNodeType::Mul:

603 case ASTNodeType::RealDiv:
604 return makeRealDiv(std::forward<T>(lhs), std::forward<U>(rhs));
605 case ASTNodeType::FloorDiv:
606 return makeFloorDiv(std::forward<T>(lhs), std::forward<U>(rhs));
607 case ASTNodeType::CeilDiv:
608 return makeCeilDiv(std::forward<T>(lhs), std::forward<U>(rhs));
609 case ASTNodeType::RoundTowards0Div:
610 return makeRoundTowards0Div(std::forward<T>(lhs), std::forward<U>(rhs));
611 case ASTNodeType::Mod:
612 return makeMod(std::forward<T>(lhs), std::forward<U>(rhs));
613 case ASTNodeType::Remainder:
614 return makeRemainder(std::forward<T>(lhs), std::forward<U>(rhs));
615 case ASTNodeType::Min:
616 return makeMin(std::forward<T>(lhs), std::forward<U>(rhs));
617 case ASTNodeType::Max:
618 return makeMax(std::forward<T>(lhs), std::forward<U>(rhs));
619 case ASTNodeType::LT:
620 return makeLT(std::forward<T>(lhs), std::forward<U>(rhs));
621 case ASTNodeType::LE:
622 return makeLE(std::forward<T>(lhs), std::forward<U>(rhs));
623 case ASTNodeType::GT:
